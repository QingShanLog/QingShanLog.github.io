4.批量处理

批量处理一组独立的模型请求可以显著提高性能并降低成本，因为处理可以并行进行。

responses = model.batch([
    "为什么鹦鹉有色彩斑斓的羽毛？",
    "飞机是怎么飞起来的？",
    "什么是量子计算？"
])
for response in responses:
    print(response)
    print("-" * 20)
    print(response.content)
    print("-" * 20)

默认情况下，batch() 仅返回整个批次的最终输出。如果您希望在每个单独的输入生成完成时接收其输出，可以使用 batch_as_completed() 来流式传输结果。

for response in model.batch_as_completed([
    "为什么鹦鹉有色彩斑斓的羽毛？",
    "飞机是怎么飞起来的？",
    "什么是量子计算？"
]):
    print("-" * 20)
    print(response)
    print("-" * 20)
questions = [
    "为什么鹦鹉有色彩斑斓的羽毛？",
    "飞机是怎么飞起来的？",
    "什么是量子计算？"
]

for idx, response in model.batch_as_completed(questions):
    print("-" * 40)
    print(f"问题: {questions[idx]}")
    print(f"回答: {response.content}")
    print("-" * 40)

batch_as_completed() 返回的结果顺序既不是正序也不是倒序，而是按照完成顺序返回。
具体来说：
完成即返回：哪个请求先得到响应，就先返回哪个结果
不保证原始顺序：结果的顺序取决于每个请求的处理速度
异步并行处理：所有请求同时发送，但根据实际响应时间依次返回

stream

1.基本用法

大多数模型可以在生成输出内容时进行流式传输。通过渐进地显示输出，流式传输显著改善了用户体验，特别是对于较长的响应。

调用stream()返回一个 迭代器，该迭代器会在输出块产生时产出它们。您可以使用循环来实时处理每个块：

for chunk in model.stream("为什么鹦鹉有色彩斑斓的羽毛？"):
    print(chunk.text, end="|", flush=True)

与 invoke() 在模型完成生成完整响应后返回单个 AIMessage 不同，stream() 返回多个 AIMessageChunk 对象，每个对象包含输出文本的一部分。重要的是，流中的每个块都旨在通过相加累积成一条完整的消息。

full = None  # None | AIMessageChunk
for chunk in model.stream("What color is the sky?"):
    full = chunk if full is None else full + chunk
    print(full.text)

# The
# The sky
# The sky is
# The sky is typically
# The sky is typically blue
# ...

print(full.content_blocks)
# [{"type": "text", "text": "The sky is typically blue..."}]

2.2 model.invoke()

LangChain 在某些情况下会自动启用流式传输模式，即使您没有显式调用流式传输方法，从而简化了聊天模型的流式处理。当您使用非流式 invoke 方法但仍希望流式传输整个应用程序（包括来自聊天模型的中间结果）时，这特别有用。
例如，在 LangGraph 智能体中，您可以在节点内调用 model.invoke()，但如果是在流式模式下运行，LangChain 将自动委托给流式传输。
当您 invoke() 一个聊天模型时，如果 LangChain 检测到您正尝试对整个应用程序进行流式传输，它将自动切换到内部流式模式。对于使用 invoke 的代码来说，调用的结果将是相同的；然而，在聊天模型进行流式传输的同时，LangChain 会负责在 LangChain 的回调系统中调用 on_llm_new_token 事件。
回调事件允许 LangGraph 的 stream() 和 astream_events() 实时呈现聊天模型的输出。

工具调用

模型可以请求调用执行任务的工具，例如从数据库中获取数据、搜索网页或运行代码。工具包含以下配对：
1.模式 (Schema)，包括工具名称、描述和/或参数定义（通常是 JSON 模式）
2.要执行的函数或协程 (coroutine)。
这是用户与模型之间基本的工具调用流程：
要使您定义的工具可供模型使用，必须使用 bind_tools 绑定它们。在随后的调用中，模型可以根据需要选择调用任何已绑定的工具。
一些模型提供商提供内置工具，可以通过模型或调用参数启用（例如 ChatOpenAI，ChatAnthropic）。详情请查看相应的提供商参考。

1.model.bind_tools(tools)

from langchain.chat_models import init_chat_model
from langchain.messages import HumanMessage, AIMessage, SystemMessage, ToolMessage
import asyncio
import os
from dotenv import load_dotenv
from langchain.tools import tool
load_dotenv()

#启动命令 uv run python D:\pythonReo\langChain_sgg\langChainModel\model.py
os.environ["OPENAI_API_KEY"] = os.getenv("OPENAI_API_KEY")
os.environ["OPENAI_BASE_URL"] = os.getenv("OPENAI_BASE_URL")
model = init_chat_model(
    model="qwen3.7-plus", # 模型名称
    model_provider="openai",  # 模型提供者
    max_tokens=1024,
)

@tool
def get_weather(location: str) -> str:
    """获取指定城市的天气信息。"""
    weather_data = {
        "西安": "西安今天阳光明媚，气温25°C，空气质量良好。",
        "北京": "北京今天多云转晴，气温18°C，有轻微雾霾。",
        "上海": "上海今天小雨，气温22°C，湿度较大。",
        "广州": "广州今天炎热，气温32°C，注意防晒。",
    }
    return weather_data.get(location, f"{location}的天气数据暂时 unavailable。")

@tool
def get_population(city: str) -> str:
    """获取指定城市的人口数量。"""
    population_data = {
        "西安": "西安市常住人口约1300万人。",
        "北京": "北京市常住人口约2200万人。",
        "上海": "上海市常住人口约2500万人。",
        "广州": "广州市常住人口约1900万人。",
    }
    return population_data.get(city, f"{city}的人口数据暂时 unavailable。")

@tool
def calculate(expression: str) -> str:
    """计算数学表达式。"""
    try:
        result = eval(expression)
        return f"计算结果: {expression} = {result}"
    except Exception as e:
        return f"计算错误: {str(e)}"

# 绑定所有工具
tools = [get_weather, get_population, calculate]
model_with_tools = model.bind_tools(tools)

def execute_tool_call(tool_call):
    """执行工具调用并返回结果。"""
    tool_name = tool_call['name']
    tool_args = tool_call['args']
    
    print(f"\n🔧 调用工具: {tool_name}")
    print(f"📋 参数: {tool_args}")
    
    if tool_name == 'get_weather':
        return get_weather.invoke(tool_args)
    elif tool_name == 'get_population':
        return get_population.invoke(tool_args)
    elif tool_name == 'calculate':
        return calculate.invoke(tool_args)
    else:
        return f"未知工具: {tool_name}"

def chat_with_tools(question: str):
    """与带工具的模型进行对话。"""
    print(f"\n{'='*50}")
    print(f"用户问题: {question}")
    print(f"{'='*50}")
    
    # 第一次调用模型
    response = model_with_tools.invoke(question)
    
    # 检查是否有工具调用
    if hasattr(response, 'tool_calls') and response.tool_calls:
        messages = [response]
        
        # 执行所有工具调用
        for tool_call in response.tool_calls:
            result = execute_tool_call(tool_call)
            print(f"✅ 工具返回: {result}\n")
            
            # 添加工具返回消息
            messages.append(
                ToolMessage(content=result, tool_call_id=tool_call['id'])
            )
        
        # 将工具结果传回模型，生成最终回答
        print("💬 生成最终回答...")
        final_response = model_with_tools.invoke(messages)
        print(f"\n🤖 最终回答: {final_response.content}")
    else:
        # 没有工具调用，直接返回模型回答
        print(f"\n🤖 模型回答: {response.content}")

6.流式工具调用

流式传输响应时，工具调用通过 ToolCallChunk 渐进式构建。这允许您在工具调用生成时查看到它们，而不是等待完整响应。

在标准的工具调用机制中，大模型的输出通道是 “互斥” 的：

  1. 不需要工具时：大模型走 content 通道，输出自然语言文本。
  2. 需要工具时：大模型走 tool_calls 通道，输出结构化的 JSON 参数。

当大模型判断“我需要查天气”时，它会把所有的算力用来生成 tool_calls（工具名和参数），此时它的 content（文本回复）字段会被置空或返回 null。 也就是说，“工具调用的 JSON 参数”本身就是它第一轮的回复，它不会一边说“好的，我来帮你查一下”，一边在后台生成 JSON。

class CustomCacheModel(ChatOpenAI):
    """自定义缓存键的模型"""
    
    def _get_prompt_cache_key(self, messages):
        """
        自定义缓存键生成逻辑
        这里我们忽略消息中的细微差异，只提取关键内容
        """
        # 提取所有消息的内容
        content = ""
        for msg in messages:
            if isinstance(msg, HumanMessage):
                content += msg.content
        
        # 标准化处理：去除标点、转小写等
        normalized = content.lower().strip()
        
        # 如果包含电影名，使用电影名作为缓存键的核心
        if "盗梦空间" in normalized or "inception" in normalized.lower():
            cache_key_content = "movie_inception"
        elif "星际穿越" in normalized or "interstellar" in normalized.lower():
            cache_key_content = "movie_interstellar"
        else:
            # 其他情况使用内容的哈希值
            cache_key_content = hashlib.md5(normalized.encode('utf-8')).hexdigest()
        
        # 添加模型信息到缓存键
        cache_key = f"{self.model_name}_{cache_key_content}"
        
        print(f"   [缓存键] {cache_key}")
        return cache_key
